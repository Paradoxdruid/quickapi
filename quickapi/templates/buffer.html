API Usage:

            import requests
import json
import urllib

conditions = {
    "buffer_conc_inital": 1,
    "buffer_conc_final": 0.1,
    "pka": 4.76,
    "volume": 1,
    "hcl": 6,
    "naoh": 6,
    "initial_ph": 4.5,
    "final_ph": 5.2, 
}

params = urllib.parse.urlencode(conditions)

r = requests.get(f"https://bonh.am/buffer?{params}&pretty=0")
recipe = r.json().get("recipe")
print(recipe) 
# "Add 0.1 liters stock buffer, 0.0063 liters of stock NaOH, and 0.8937 liters of water."
